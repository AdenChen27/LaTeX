% Theorem style macro
\newcommand{\getThmboxStyleA}[4]{%
  \tcbset{%
    #2/.style={
      ThmBoxBaseStyle,
      colframe=#3,
      colback=#4
    }
  }%
  \declaretheoremstyle[
    headfont=\bfseries\color{#3},
    notefont=\bfseries\color{#3},
    bodyfont=\normalfont,
    spaceabove=5pt,
    spacebelow=5pt,
    headpunct={. },
    postheadspace=0pt,
    theoremstyle=tcb,
    tcbstyle=#2
  ]{#1}%
}
