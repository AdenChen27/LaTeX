% \makepsethead[due date]{course}{assignment}{author}
% Renders a centered header block and TOC. Call at the top of \begin{document}.
% Compiled date is automatic (\today). Omit due date to hide it.
% Example: \makepsethead[2026-04-30]{MATH 101}{Problem Set 3}{John Smith}
\newcommand{\makepsethead}[4][]{%
  \begin{center}
    {\large\scshape #2}\par
    \vspace{1em}
    {\LARGE\bfseries #3}\par
    \vspace{1ex}
    {#4}\par
    \smallskip
    {\footnotesize Compiled:~\today}\par
  \end{center}
  \vspace{0.5ex}
  \noindent\rule{\linewidth}{0.4pt}\par
  \def\@psetdue{#1}%
  \ifx\@psetdue\@empty\else
    {\footnotesize\itshape Due:~#1}\par
  \fi
  \par\vspace{2\bigskipamount}
  \begingroup
    \let\clearpage\relax
    \@starttoc{toc}%
  \endgroup
}
