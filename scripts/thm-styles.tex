% Base tcolorbox options shared by all theorem boxes.
% Uses only standard tcolorbox options (no enhanced/skins) for Beamer compatibility.
% leftrule=3pt + colframe (set per style) replaces enhanced+frame hidden+borderline west.
\tcbset{
  adenc thmboxbase/.style = {
    breakable,
    arc = 0pt,
    leftrule = 3pt,
    rightrule = 0pt,
    toprule = 0pt,
    bottomrule = 0pt,
    left = .75em,
    right = 0pt,
    top = 3pt,
    bottom = 5pt,
    before skip = 5pt,
    after skip = .5em,
  }
}

% Creates a thmtools style (head formatting only) and a named tcolorbox style.
% (usage: \getThmboxStyle{style name}{foreground color}{background color}{body font})
\newcommand{\getThmboxStyle}[4]{
  \declaretheoremstyle[
    headfont = \bfseries\color{#2},
    notefont = \bfseries\color{#2},
    bodyfont = #4,
    headpunct = {. },
    postheadspace = {0pt},
  ]{#1}
  \tcbset{adenc #1/.style = {adenc thmboxbase, colback=#3, colframe=#2}}
}

% If theorembox=plain|color: use tcolorbox-boxed styles (breakable across pages), no end symbols
% Default (theorembox=none): use default styles `plain`, `definition`, and `remark`, with end symbols
\adencIfBoxed{%
  \newcommand{\thmboxStyle}{thmbox}
  \newcommand{\defboxStyle}{defbox}
  \newcommand{\exboxStyle}{exbox}
  \newcommand{\probboxStyle}{probbox}
  % No end symbols for example, remark, and problem
  \newcommand\remqed{}
  \newcommand\exqed{}
  \newcommand\probqed{}
}{%
  \newcommand{\thmboxStyle}{plain}
  \newcommand{\defboxStyle}{definition}
  \newcommand{\exboxStyle}{remark}
  \newcommand{\probboxStyle}{remark}
  \adencIfPset{%
    \declaretheoremstyle[
      headfont = \bfseries\color{colorpset},
      notefont = \bfseries\color{colorpset},
      bodyfont = \normalfont\color{colorpset},
      headpunct = {. },
      postheadspace = {0pt},
    ]{psetprobbox}
    \renewcommand{\probboxStyle}{psetprobbox}
  }{}
  % End symbols for example and remark
  \adencIfFancyqed{%
    \newcommand\remqed{\Coffeecup}

  }{%
    \newcommand\remqed{\(\diamond\)}
    \newcommand\exqed{\(\triangleleft\)}
    \newcommand\probqed{\(\triangle\)}
  }
}

% Apply tcolorbox wrapping for boxed mode.
% Defined outside \adencIfBoxed so expl3 catcodes are set before tokenisation.
\ExplSyntaxOn
\cs_new_protected:Nn \__adenc_apply_thmboxes:
{
  \clist_map_inline:nn {theorem, lemma, proposition, corollary}
  {
    \tcolorboxenvironment{##1}{adenc~thmbox}
    \tcolorboxenvironment{##1*}{adenc~thmbox}
  }
  \clist_map_inline:nn {definition}
  {
    \tcolorboxenvironment{##1}{adenc~defbox}
    \tcolorboxenvironment{##1*}{adenc~defbox}
  }
  \clist_map_inline:nn {example}
  {
    \tcolorboxenvironment{##1}{adenc~exbox}
    \tcolorboxenvironment{##1*}{adenc~exbox}
  }
  \clist_map_inline:nn {problem, exercise, question}
  {
    \tcolorboxenvironment{##1}{adenc~probbox}
    \tcolorboxenvironment{##1*}{adenc~probbox}
  }
}
\cs_new_protected:Nn \__adenc_apply_pset_thmboxes:
{
  \clist_map_inline:nn {problem, exercise, question}
  {
    \tcolorboxenvironment{##1}{adenc~psetprobbox}
    \tcolorboxenvironment{##1*}{adenc~psetprobbox}
  }
}
\cs_new_protected:Nn \adenc_apply_thmboxes_bridge:
{
  \adencIfBoxed { \__adenc_apply_thmboxes: } { \adencIfPset { \__adenc_apply_pset_thmboxes: } { } }
}
\adenc_apply_thmboxes_bridge:
\ExplSyntaxOff
