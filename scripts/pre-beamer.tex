\ExplSyntaxOn

\bool_new:N \l_adenc_color_bool        \bool_set_false:N \l_adenc_color_bool
\bool_new:N \l_adenc_boxed_bool        \bool_set_true:N  \l_adenc_boxed_bool  % boxed by default in beamer
\bool_new:N \l_adenc_hideproofs_bool   \bool_set_false:N \l_adenc_hideproofs_bool
\bool_new:N \l_adenc_hidemarkings_bool \bool_set_false:N \l_adenc_hidemarkings_bool

\keys_define:nn { adenc-beamer } {
  theorembox .choices:nn = { none, plain, color } {
    \int_case:nn { \l_keys_choice_int } {
      { 1 } { \bool_set_false:N \l_adenc_boxed_bool  \bool_set_false:N \l_adenc_color_bool }
      { 2 } { \bool_set_true:N  \l_adenc_boxed_bool  \bool_set_false:N \l_adenc_color_bool }
      { 3 } { \bool_set_true:N  \l_adenc_boxed_bool  \bool_set_true:N  \l_adenc_color_bool }
    }
  },
  theorembox .default:n = none,
  hideproofs   .bool_set:N = \l_adenc_hideproofs_bool,   hideproofs   .default:n = true,
  hidemarkings .bool_set:N = \l_adenc_hidemarkings_bool, hidemarkings .default:n = true,
}

\ProcessKeysOptions { adenc-beamer }
