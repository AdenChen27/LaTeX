\keys_define:nn { adenc } {
  % theorembox: choice key controlling box style and color
  % none  = plain amsthm styles, no color (default)
  % plain = tcolorbox boxes, no color
  % color = tcolorbox boxes + color
  theorembox .choices:nn = { none, plain, color } {
    \int_case:nn { \l_keys_choice_int } {
      { 1 } { \bool_set_false:N \l_adenc_boxed_bool  \bool_set_false:N \l_adenc_color_bool }
      { 2 } { \bool_set_true:N  \l_adenc_boxed_bool  \bool_set_false:N \l_adenc_color_bool }
      { 3 } { \bool_set_true:N  \l_adenc_boxed_bool  \bool_set_true:N  \l_adenc_color_bool }
    }
  },
  theorembox .default:n = none,
  % \usepackage[hideproofs]{adenc} to hide everything in the proof environment.
  hideproofs   .bool_set:N = \l_adenc_hideproofs_bool,   hideproofs   .default:n = true,
  % \usepackage[hidemarkings]{adenc} to hide all \markabove and \markbelow annotations.
  hidemarkings .bool_set:N = \l_adenc_hidemarkings_bool, hidemarkings .default:n = true,
  % \usepackage[workingpaper]{adenc} for wider margins (todonotes) and draft watermark.
  workingpaper .bool_set:N = \l_adenc_workingpaper_bool, workingpaper .default:n = true,
  % \usepackage[index]{adenc} to enable automatic indexing of \vocab terms.
  index        .bool_set:N = \l_adenc_index_bool,        index        .default:n = true,
  % \usepackage[fancyqed=false]{adenc} to use the default qed symbol.
  fancyqed     .bool_set:N = \l_adenc_fancyqed_bool,     fancyqed     .default:n = true,
  % \usepackage[header=false]{adenc} to disable the running section header.
  header       .bool_set:N = \l_adenc_header_bool,       header       .default:n = true,
  % \usepackage[clearsection=false]{adenc} to disable new-page at each section.
  clearsection .bool_set:N = \l_adenc_sectionpage_bool,  clearsection .default:n = true,
}

% Default: running section header on left, centered page number in footer.
% Override header content per-document: \renewcommand{\adencheadercontent}{...}
% Disable entirely: \usepackage[header=false]{adenc}
\newcommand{\adencheadercontent}{\small\itshape\nouppercase{\leftmark}}
\adencIfHeader{%
  \RequirePackage{fancyhdr}
  % Use AtBeginDocument so our setup runs after amsart resets the page style.
  \AtBeginDocument{%
    \pagestyle{fancy}
    \fancyhf{}
    \fancyhead[L]{\adencheadercontent}
    \renewcommand{\headrulewidth}{0.4pt}
    \fancyfoot[C]{\small\thepage}
    \renewcommand{\sectionmark}[1]{\markboth{\thesection\enspace#1}{}}
    % plain style: title page gets no header, but keeps centered page number
    \fancypagestyle{plain}{%
      \fancyhf{}%
      \fancyfoot[C]{\small\thepage}%
      \renewcommand{\headrulewidth}{0pt}%
    }
    % Clear amsart's author/title marks so the header is empty until the first section.
    \markboth{}{}%
  }%
}{}

% sectionpage: new page + suppress header at each section (covers \section and \section*)
\adencIfSectionpage{%
  \AddToHook{cmd/section/before}{\clearpage}
  \adencIfHeader{%
    \AddToHook{cmd/section/before}{\thispagestyle{plain}}
  }{}
}{}
