% Default: running section header on left, centered page number in footer.
% Override header content per-document: \renewcommand{\adencheadercontent}{...}
% Disable entirely: \usepackage[noheader]{adenc}
\newcommand{\adencheadercontent}{\small\itshape\nouppercase{\leftmark}}
\ifadencheader
  \RequirePackage{fancyhdr}
  \pagestyle{fancy}
  \fancyhf{}
  \fancyhead[L]{\adencheadercontent}
  \renewcommand{\headrulewidth}{0.4pt}
  \fancyfoot[C]{\small\thepage}
  \renewcommand{\sectionmark}[1]{\markboth{\thesection\enspace#1}{}}
  % plain style: title page gets no header, but keeps centered page number
  \fancypagestyle{plain}{%
    \fancyhf{}%
    \fancyfoot[C]{\small\thepage}%
    \renewcommand{\headrulewidth}{0pt}%
  }
\fi

% sectionpage: new page + suppress header at each section (covers \section and \section*)

  \AddToHook{cmd/section/before}{\clearpage}
  \ifadencheader
    \AddToHook{cmd/section/before}{\thispagestyle{plain}}
  \fi
\fi
